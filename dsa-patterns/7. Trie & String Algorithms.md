```cpp
void insert(TrieNode* root, string word) {
    TrieNode* node = root;
    for (char c : word) {
        int i = c - 'a';
        if (!node->children[i]) node->children[i] = new TrieNode();
        node = node->children[i];
    }
    node->isEnd = true;
}
```

âœ… Search for Word or Prefix

```cpp
bool search(TrieNode* root, string word, bool fullWord = true) {
    TrieNode* node = root;
    for (char c : word) {
        int i = c - 'a';
        if (!node->children[i]) return false;
        node = node->children[i];
    }
    return fullWord ? node->isEnd : true;
}
```

### 5. KMP Prefix Table Template

Used in problems like LC 28 (Implement strStr) or detecting repeating patterns.

âœ… Prefix Table (Failure Array)

```cpp
vector<int> computeLPS(string& s) {
    int n = s.size(), len = 0;
    vector<int> lps(n);
    for (int i = 1; i < n; i++) {
        while (len > 0 && s[i] != s[len])
            len = lps[len - 1];
        if (s[i] == s[len]) len++;
        lps[i] = len;
    }
    return lps;
}
```
